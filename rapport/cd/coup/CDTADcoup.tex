\begin{algorithme}
  \begin{enregistrement}{\textbf{Coup}}
    \champEnregistrement{pion}{\textbf{Pion}}
    \champEnregistrement{position}{\textbf{Position}}
  \end{enregistrement}
\end{algorithme}

\vspace*{5mm}

\begin{algorithme}
  \small
  \fonction
      {coupCoup}
      {pion :\textbf{Pion}, pos : \textbf{Position}}
      {\textbf{Coup}}
      {}
      {
        \affecter{coup.position}{pos}
        \affecter{coup.pion}{pion}
        \retourner{coup}
      }
      
\end{algorithme}
