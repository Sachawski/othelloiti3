\begin{algorithme}

  \fonction
      {compteurCouleur}
      {plateau : Plateau, couleur : Couleur}
      {Naturel}
      {res : Naturel, i,j : 1..8, pos : Position}
      {\affecter{res}{0}
        \pour{i}{1}{8}{}
             {\pour{j}{1}{8}{}
               {\affecter{pos}{position(i,j)}
                \sialors{non(estCaseVide(pos,plateau)) et (obtenirCouleurSuperieur(obtenirPion(pos,plateau)))}
                             {\affecter{compteur}{compteur+1}}
               }
             }
      }
         
\end{algorithme}
