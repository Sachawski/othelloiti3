\begin{algorithme}
  \small
  \fonction{plateauBloque}{plateau:Plateau}{\booleen}
           {aucunePositionPossible : \booleen,coup : Coup, i,j : \Naturel,pos : Position}
           {\affecter{aucunePositionPossible}{Faux}
             \pour{i}{1}{8}{}{
               \pour{j}{1}{8}{}{
                 \affecter{pos}{position(i,j)}
                 \affecter{coup}{coupCoup(obtenirPion(pos,plateau),pos)}
                \sialors{cardinalite(prisAuPiege(plateau,coup)) $\neq$ 0}{\retourner{Faux}}}}
                \retourner{Vrai}}
\end{algorithme}
