\begin{algorithme}
  \small
  \fonction
  {initialiserPlateau}
  {}
  {Plateau}
  {p:Plateau\\
  pPion:Pion\\
  i,j:\naturel\\
  pos:Position}
  { 
    \affecter{p}{plateau()}
    \pour{i}{4}{5}{}
    {
      \pour{j}{4}{5}{}
      {
        \sialorssinon{i=j}
        {
          \affecter{pPion}{creerPion(blanc())}
          \affecter{pos}{position(i,j)}\\
          poserPion(pos,pPion,p)
        }
        {
          \affecter{pPion}{creerPion(noir())}
          \affecter{pos}{position(i,j)}\\
          poserPion(pos,pPion,p)
        }
      }
    }
    \retourner{p}
  }
\end{algorithme}
