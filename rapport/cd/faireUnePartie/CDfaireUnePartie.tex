\begin{algorithme}
  \small
  \fonction
  {FaireUnePartie}
  {couleurJoueur: Couleur, obtenirLeCoup: \typePointeur{fonction}, afficherPlateau: \typePointeur{procedure}, obtenirCoups :\typePointeur{fonction}}
  {Couleur}
  {pjouer: Pion\\
  pIA: Pion\\
  coupjouer: Coup\\
  coupIA: Coup\\
  auJouee: \booleen\\
  auIA: \booleen\\
  pos:Position\\
  x,y:[1..8][1..8]\\
  }
  { 
    \affecter{pIA}{creerPion(changeCouleur(couleur))}
    \affecter{pjouee}{creerPion(couleurJoueur)}
    \affecter{p}{initialiserPlateau()}
    \sialorssinon{couleurJoueur=noir()}
    { \affecter{auJouee}{VARI}
      \affecter{auIA}{FAUX}
    }
    { \affecter{auIA}{VRAI}
      \affecter{auJouee}{FAUX}
    }\\
    afficherPlateau(p)
    \tantque{non(verifierFini(p))}
    {
      \sialorssinon{auJouee}
      {
        \affecter{coupjouer}{obtenirLeCoup()}
        \affecter{auIA}{non(auIA)}
        \affecter{auJouee}{non(auJouee)}\\
        Jouer(p,coupjouer)
      }
      {
        \affecter{coupIA}{obtenirCoups(p)}
        \affecter{auIA}{non(auIA)}
        \affecter{auJouee}{non(auJouee)}\\
        Jouer(p,coupIA)
      }\\
      afficherPlateau(p)
    }
    \retourner {Resultat(p)}
  }
\end{algorithme}
