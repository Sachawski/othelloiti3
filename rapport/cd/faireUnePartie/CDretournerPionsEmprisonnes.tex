\begin{algorithme}
\procedure{retournerPionEmprisonnes}{\paramEntreeSortie{plateau : Plateau} , \paramEntree{coup : Coup}}
{coupTemp : Coup , x, y ,directionX, directionY, i :\naturel, coupsEmprisonnants : Coups, pos : Position}
{
\affecter{coupsEmprisonnants}{pionMemeCouleur(plateau,coup,coupsAdjacentsAdversaires(plateau,coup))}
\pour{i}{1}{nbCoups(coupsEmprisonnants)}{}{
{\affecter{coupTemp}{iemeCoup(coupsEmprisonnant,i)}
\affecter{x}{obtenirX(obtenirPositionCoup(coupTemp))}
\affecter{y}{obtenirY(obtenirPositionCoup(coupTemp))}
\sialorssinon{x-obtenirX(coup)==0}{\affecter{directionX}{0}
\affecter{directionY}{$\frac{y-obtenirY(coup)}{abs(y-obtenirY(coup))}$}}
{\sialorssinon{y-obtenirY(coup)==0}{\affecter{directionY}{0}
\affecter{directionX}{$\frac{x-obtenirX(coup)}{abs(x-obtenirX(coup))}$}}{
\affecter{directionX}{$\frac{x-obtenirX(coup)}{abs(x-obtenirX(coup))}$}
\affecter{directionY}{$\frac{y-obtenirY(coup)}{abs(y-obtenirY(coup))}$}
}
}
\tantque{coupObtenirPositionCoup(coupTemp) $\neq$ coupObtenirPositionCoup(coup)}
{	
	\affecter{x}{x+directionX}
	\affecter{y}{y+directionY}
	\sialors{x$\geq$1 et x$\leq$8 et y$\geq$1 et y$\leq$8}{
	\affecter{pos}{position(x,y)}
	\affecter{coupObtenirPositionCoup(coupTemp)}{pos)}
	\sialors{obtenirCouleurSuperieure(coupObtenirPionCoup(coupTemp))$\neq$ obtenirCouleurSuperieure(coupObtenirPionCoup(coup)}
	{\instruction{Pion.retournerPion(coupObtenirPionCoup(coupTemp))}}
	
	}
	
}
}
}
}
\end{algorithme}
